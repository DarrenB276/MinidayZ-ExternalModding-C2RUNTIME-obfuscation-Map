  var rt = window.cr_getC2Runtime();  // get runtime
  rt.wa                               // current layout (cr: rt.running_layout)
  rt.wa.ua                            // layers array (cr: rt.running_layout.layers)
  rt.wa.ua[0].q                       // layer 0 instances (cr: ...layers[0].instances)
  rt.S                                // types_by_index
  rt.S[i].q                           // all instances of type i
  rt.jg                               // objectsByUid map
  rt.jg[uid]                          // get instance by UID
