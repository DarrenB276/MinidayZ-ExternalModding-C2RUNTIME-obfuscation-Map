--- Как вызвать системное действие из обфусцированного кода ---
  Пример: Установить переменную
    H.prototype.n.GC(...)             // system_object.acts.SetVar

--- Как читать/записывать свойства экземпляра ---
  inst.x, inst.y                      // позиция (то же имя)
  inst.width, inst.height             // размер (то же имя)
  inst.u                              // угол (cr: inst.angle)
  inst.visible                        // видимость (то же имя)
  inst.opacity                        // непрозрачность (то же имя)
  inst.B                              // массив свойств (cr: inst.properties)
  inst.cc                             // instance_vars (cr: inst.instance_vars)
  inst.C                              // ссылка на слой (cr: inst.layer)
  inst.ka                             // bbox (cr: inst.bbox)
  inst.P()                            // пометить bbox изменённым (cr: inst.set_bbox_changed())

--- Как получить ссылки на runtime ---
  var rt = window.cr_getC2Runtime();  // получить runtime
  rt.wa                               // текущий макет (cr: rt.running_layout)
  rt.wa.ua                            // массив слоёв (cr: rt.running_layout.layers)
  rt.wa.ua[0].q                       // экземпляры слоя 0 (cr: ...layers[0].instances)
  rt.S                                // types_by_index
  rt.S[i].q                           // все экземпляры типа i
  rt.jg                               // карта objectsByUid
  rt.jg[uid]                          // получить экземпляр по UID

--- Как вызывать события ---
  rt.trigger(H.prototype.k.METHOD, inst, type)

--- Как создавать экземпляры ---
  rt.Ut(type, layer, x, y, createRow)  // createInstance

--- Как сменить макет ---
  rt.Mh = rt.qb[layoutIndex]           // установить changelayout

--- Как получить доступ к методам плагина ---
  // Получить запись objectRefTable
  var ref = rt.ya(index);              // GetObjectReference
  // Или напрямую: rt.wy[index]
